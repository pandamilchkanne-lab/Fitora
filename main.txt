<script>
  /* ── STORAGE KEYS ─────────────────────────────── */
  const KEY = 'myapp_profile';

  /* ── LOAD DATA ────────────────────────────────── */
  function loadData() {
    try { return JSON.parse(localStorage.getItem(KEY)) || {}; }
    catch { return {}; }
  }

  function saveData(data) {
    localStorage.setItem(KEY, JSON.stringify(data));
  }

  /* ── RENDER PROFILE ───────────────────────────── */
  function renderProfile() {
    const d = loadData();

    const firstname = d.firstname || '';
    const lastname  = d.lastname  || '';
    const fullname  = [firstname, lastname].filter(Boolean).join(' ');
    const initials  = (firstname[0] || '') + (lastname[0] || '');

    // Avatar
    document.getElementById('avatarInitials').textContent = initials || '?';
    document.getElementById('avatarName').textContent     = fullname || 'Dein Name';
    document.getElementById('avatarJob').textContent      = d.job    || 'Keine Angabe';

    // Format birthday
    let birthDisplay = '—';
    if (d.birth) {
      const [y, m, day] = d.birth.split('-');
      birthDisplay = `${day}.${m}.${y}`;
    }

    set('d-name',        fullname,      '—');
    set('d-birth',       birthDisplay,  '—');
    set('d-location',    d.location,    '—');
    set('d-nationality', d.nationality, '—');
    set('d-email',       d.email,       '—');
    set('d-phone',       d.phone,       '—');
    set('d-website',     d.website,     '—');
    set('d-job',         d.job,         '—');
    set('d-bio',         d.bio,         '—');
    set('d-interests',   d.interests,   '—');
  }

  function set(id, val, fallback) {
    const el = document.getElementById(id);
    if (!el) return;
    const v = val || fallback;
    el.textContent = v;
    el.classList.toggle('placeholder', !val);
  }

  /* ── MODAL LOGIC ──────────────────────────────── */
  function openModal(section) {
    const d = loadData();
    const overlay = document.getElementById('modal-' + section);

    if (section === 'personal') {
      document.getElementById('f-firstname').value   = d.firstname   || '';
      document.getElementById('f-lastname').value    = d.lastname    || '';
      document.getElementById('f-birth').value       = d.birth       || '';
      document.getElementById('f-location').value   = d.location    || '';
      document.getElementById('f-nationality').value = d.nationality || '';
    } else if (section === 'contact') {
      document.getElementById('f-email').value   = d.email   || '';
      document.getElementById('f-phone').value   = d.phone   || '';
      document.getElementById('f-website').value = d.website || '';
    } else if (section === 'bio') {
      document.getElementById('f-job').value       = d.job       || '';
      document.getElementById('f-bio').value       = d.bio       || '';
      document.getElementById('f-interests').value = d.interests || '';
    }

    overlay.classList.add('open');
  }

  function closeAllModals() {
    document.querySelectorAll('.modal-overlay').forEach(m => m.classList.remove('open'));
  }

  /* ── SAVE SECTION ─────────────────────────────── */
  function saveSection(section) {
    const d = loadData();

    if (section === 'personal') {
      d.firstname   = document.getElementById('f-firstname').value.trim();
      d.lastname    = document.getElementById('f-lastname').value.trim();
      d.birth       = document.getElementById('f-birth').value;
      d.location    = document.getElementById('f-location').value.trim();
      d.nationality = document.getElementById('f-nationality').value.trim();
    } else if (section === 'contact') {
      d.email   = document.getElementById('f-email').value.trim();
      d.phone   = document.getElementById('f-phone').value.trim();
      d.website = document.getElementById('f-website').value.trim();
    } else if (section === 'bio') {
      d.job       = document.getElementById('f-job').value.trim();
      d.bio       = document.getElementById('f-bio').value.trim();
      d.interests = document.getElementById('f-interests').value.trim();
    }

    saveData(d);
    renderProfile();
    closeAllModals();
    showToast('✓ Gespeichert');
  }

  /* ── TOAST ────────────────────────────────────── */
  function showToast(msg) {
    const t = document.getElementById('toast');
    t.textContent = msg;
    t.classList.add('show');
    setTimeout(() => t.classList.remove('show'), 2500);
  }

  /* ── PAGE NAVIGATION ──────────────────────────── */
  function showPage(id) {
    document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
    document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
    document.getElementById('page-' + id).classList.add('active');
    document.getElementById('nav-'  + id).classList.add('active');
  }

  /* ── CLOSE MODAL ON OVERLAY CLICK ────────────── */
  document.querySelectorAll('.modal-overlay').forEach(overlay => {
    overlay.addEventListener('click', e => {
      if (e.target === overlay) closeAllModals();
    });
  });

  /* ── INIT ─────────────────────────────────────── */
  renderProfile();
</script>
